\section{Classification}
\subsection{Estimer l'efficacité d'un classifieur}
\subsubsection{K-Fold cross-validation}
\begin{frame}
  \frametitle{K-Fold cross-validation}
  \begin{itemize}
    \item{Divise les données en $k$ sections}
    \item{$k$ itérations}
    \item{$k-1$ sections d'apprentissage}
    \item{$1$ section de test}
    \item{Résultat: moyenne des taux de reconnaissances et écart-type}
    \item{Ici, $k = 10$}
  \end{itemize}
\end{frame}

\subsubsection{Courbe ROC}
\begin{frame}
  \frametitle{Courbe ROC}
  \begin{itemize}
    \item{Se base sur le taux de certitude de chaque point reconnu}
    \item{Fonction du taux de vrais positifs et de faux positifs}
  \end{itemize}
  \begin{figure}[ht]

      \caption{Courbe ROC pour plusieurs classifieurs, sur donut.}
      \label{fig:roc}
    \end{center}
  \end{figure}
\end{frame}

\subsection{Classification de données discrètes}
\subsubsection{Naïve Bayes classifier}
\begin{frame}
  \frametitle{Naïve Bayes classifier}
  \begin{itemize}
    \item{Données à caractéristiques indépendantes}
    \item{Estimer certaines probabilités à partir des données
      d'apprentissage ($P(C = c) et P(Xi = xi | C)$)}
    \item{Maximiser $P(C = c|X = x)$}
  \end{itemize}
\end{frame}

\subsubsection{Continuousification}
\begin{frame}
  \frametitle{Continuousification}
  \begin{itemize}
    \item{Rendre des données discrètes continues (pour utiliser KNN)}
    \item{Conserver la dépendance entre les variables}
    \item{Naïf: Une valeur arbitraire par valeur observée}
    \item{NBF: Une dimension par valeur observée}
    \item{VDM/MDV: Estimer et utiliser les probabilité conditionnelles}
  \end{itemize}
\end{frame}

\section{Clustering}
\subsection{Estimer l'efficacité du clustering}
\begin{frame}
  \frametitle{Estimer l'efficacité du clustering}
  \begin{itemize}
    \item{Détecter la convergence vers un extrémum local}
    \item{Somme des distances intra-cluster}
    \item{Vitesse de convergence: nombre d'itérations}
  \end{itemize}
\end{frame}

\subsection{Méthodes de clustering}
\subsubsection{K-MEANS}
\begin{frame}
  \frametitle{K-MEANS}
  \begin{itemize}
    \item{Choisir $k$ centres au hasard}
    \item{Calculer les clusters et les nouveaux centres}
    \item{Itérer jusqu'à la convergence}
  \end{itemize}
\end{frame}
